# La fonction densité theorique est 
# la fonction g(x)=exp(-x ^2/2)/sqrt(2*pi) 
g = function(x){ exp(-x^2/2)/sqrt(2*pi) } 
# Cette fonction étant tres utilisée en statistiques,
# sous R, g(x)  s'appelle dnorm(x)
# Ajout du graphe de la densité en pointillés

y = g(x)

#   Graphes mathématiques (plot classique en ligne continue)
f = function(x,m,s){ exp(-(x - m)^2/(2*s^2))/(s*sqrt(2*pi)) }
# Cette fonction etant tres utilisée en statistiques,
# sous R, f(x,m,s)  s'appelle dnorm(x, mean=m, sd=s)
x = seq(-3, 3, by=0.05) # x est ordonné
y = f(x,0,1)
z = f(x,1,1)
w = f(x,0,2)
plot(x, y, type="l", xlim=c(-3,3), ylim=c(0,0.5), col="blue", ylab="")
#   Ajout de 2 autres graphes
points(x, z, type="l", col="gréen")
points(x, w, type="l", col="red")
title("Trois gaussiennes")
legend("topleft", legend=c("(m,s)=(0,1)", "(m,s)=(1,1)", "(m,s)=(0,2)"), lty=1, col=c("blue", "gréen", "red"))
